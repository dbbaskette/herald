- [ ] **Step 3: Implement AgentFactory**

```java
package com.herald.agent;

import com.herald.agent.profile.AgentProfile;
import com.herald.tools.FileSystemTools;
import com.herald.tools.HeraldShellDecorator;
import com.herald.tools.WebTools;
import org.springframework.ai.chat.client.ChatClient;
import org.springframework.ai.chat.model.ChatModel;

import java.util.ArrayList;
import java.util.List;
import java.util.Set;

/**
 * Builds a configured {@link ChatClient} from an {@link AgentProfile} and system prompt.
 * Usable both inside and outside the Spring DI container.
 */
public final class AgentFactory {

    private AgentFactory() {}

    /**
     * Build a ChatClient from an AgentProfile record.
     *
     * @param profile     parsed agent configuration
     * @param systemPrompt markdown body from the agents.md file
     * @param chatModel   the ChatModel to use
     * @return configured ChatClient ready for use
     */
    public static ChatClient fromProfile(AgentProfile profile, String systemPrompt,
                                          ChatModel chatModel) {
        var builder = ChatClient.builder(chatModel)
                .defaultSystem(systemPrompt);

        List<Object> tools = resolveTools(profile.tools());
        if (!tools.isEmpty()) {
            builder.defaultTools(tools.toArray());
        }

        List<org.springframework.ai.chat.client.advisor.api.Advisor> advisors = new ArrayList<>();
        advisors.add(new DateTimePromptAdvisor(
                java.time.ZoneId.of("America/New_York"),
                java.time.format.DateTimeFormatter.ofPattern("EEEE, MMMM d, yyyy 'at' h:mm a z")));

        if (profile.contextFile() != null) {
            advisors.add(new ContextMdAdvisor(java.nio.file.Path.of(profile.contextFile())));
        }

        if (!advisors.isEmpty()) {
            builder.defaultAdvisors(advisors.toArray(
                    new org.springframework.ai.chat.client.advisor.api.Advisor[0]));
        }

        return builder.build();
    }

    private static List<Object> resolveTools(List<String> toolNames) {
        if (toolNames == null || toolNames.isEmpty()) return List.of();

        Set<String> requested = Set.copyOf(toolNames);
        List<Object> tools = new ArrayList<>();

        if (requested.contains("filesystem")) tools.add(new FileSystemTools());
        if (requested.contains("web")) tools.add(new WebTools(""));

        return tools;
    }
}
```

Note: This is a minimal initial implementation. `resolveTools` maps tool names to instances for stateless tools only (filesystem, web). Shell and other tools require configuration and will be added as needed. The factory is intentionally simple — it builds the ChatClient with the profile's tools and core advisors.
